S4.6.1 Overview

  • "templated" means we take a global or member function and turns it into a template
  • to turn a member f(x) into a template it's class also has to be a template
  • specified by template <typename T>
    • By convention we use an uppercase

Example: Creating a function template

int max(int, int, int);

int main() {
    
    
    return 0;
}


// What if we wanted to change this to use floats instead of ints
// The logic will still remain the same, so we want to turn it into a f(x) template
int max (int v1, int v2, int v3) {
    int maxValue = v1;
    if (v2 > maxValue) {
        maxValue = v2;
    }

    if (v3 > maxValue) {
        maxValue = v3;
    }

    return maxValue;
}


int main() {
    cout << "max int: " << max(33, 22, 44) << endl;
    cout << "max float: " << max(3.14f, 9.99f, 3.45f) << endl; // Since we have templates now, we can use any types such as floats
    cout << "max float: " << max('Z', 'z', 'c') << endl; // Or even strings

    Time t1(95);
    Time t2(13);
    Time t3(76);

    // We can now even use it with our own objects
    cout << "Max time: " << max(t1, t2, t3) << endl;
}

template <typename T>
T max (T v1, T v2, T v3) { // We change the int's to use typename T instead
    T maxValue = v1;
    if (v2 > maxValue) {
        maxValue = v2;
    }

    if (v3 > maxValue) {
        maxValue = v3;
    }

    return maxValue;
}
